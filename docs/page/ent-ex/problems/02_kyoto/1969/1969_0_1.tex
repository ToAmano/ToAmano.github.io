　次の各式は三角関数についての公式である．
(1)〜(4)から，(5)〜(7)を導け．
\begin{description}
\item[(1)]$\sin(-\alpha)=-\sin\alpha$
\item[(2)]$\cos(-\alpha)=\cos\alpha$
\item[(3)]$\sin(\alpha+\beta)=\sin\alpha\cos\beta+\cos\alpha\sin\beta$
\item[(4)]$\cos(\alpha+\beta)=\cos\alpha\cos\beta-\sin\alpha\sin\beta$
\item[(5)]$\displaystyle\tan(\alpha+\beta)=\frac{\tan\alpha+\tan\beta}{1-\tan\alpha\tan\beta}$
\item[(6)]$\displaystyle\sin\alpha+\sin\beta=2\sin\frac{\alpha+\beta}{2}\cos\frac{\alpha-\beta}{2}$
\item[(7)]$\displaystyle\cos\alpha+\cos\beta=2\cos\frac{\alpha+\beta}{2}\cos\frac{\alpha-\beta}{2}$
\end{description}
\end{flushleft}
