\item[(i)]$m$，$n$は自然数とする．三角関数の加法定理を用いて，等式
\[ \sin mx \sin nx = \frac{1}{2} \{ \cos(m-n)x-\cos(m+n)x \} \]
が成り立つことを示し，さらに次の積分$I_{m,n}$を求めよ．
\[ I_{m,n}=\int_{-\pi}^\pi \sin mx \sin nx dx \]
\item[(ii)]整数$k$ $(0 \leqq k \leqq 5)$，自然数$m$，$n$および実数$a$，$b$に対して，
\[ f(k)=\int_{-\pi}^\pi{(\sin kx - a \sin mx - b \sin nx)}^2 dx, \hspace{1zw}
p(k)=\frac{5 \, !}{k \, ! (5-k) \, !} {\left( \frac{1}{2} \right)}^5, \hspace{1zw}
E=\sum_{k=0}^5 p(k)f(k) \]
とおくとき，$E$を最小にするような$m$，$n$，$a$，$b$を求めよ．
\end{description}
