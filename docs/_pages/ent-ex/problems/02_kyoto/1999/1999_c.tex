　$\alpha$を正の定数として，数列$a_n$，$b_n$ $(n\geqq1)$を次の式で定める．
\begin{description}
\item[]$2a_{n+1}=\alpha(3{a_n}^2+2a_nb_n-{b_n}^2-a_n+b_n)$
\item[]$2b_{n+1}=\alpha(-{a_n}^2-2a_nb_n-{b_n}^2-a_n+b_n)$
\item[]$a_1=b_1=1$
\end{description}
\begin{description}
\item[(1)]$a_2$，$b_2$，$a_3$，$b_3$，$a_4$，$b_4$を求めよ．
\item[(2)]$\displaystyle\frac{a_{2n+1}}{a_{2n}}$を求めよ．
\end{description}
