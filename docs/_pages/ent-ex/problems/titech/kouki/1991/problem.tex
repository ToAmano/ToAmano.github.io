    \begin{enumerate}
        \item $a_n$を求めよ．
        \item 上の数のうちで，1の位の数字が0である数の個数を$b_n$とするとき，$ \lim_{n \to \infty} \frac{b_n}{a_n}$を求めよ．
    \end{enumerate}
